Pipes| | | |

course object properties: title, rating, students, price, releaseDate
Search angular.io to get information about each pipe:
https://angular.io/api/common/DecimalPipe
https://angular.io/api/common/DatePipe

Title: '{{ course.title }}'
course.title | uppercase = ' {{ course.title | uppercase }}'
course.title | lowercase = ' {{ course.title | lowercase }}'
course.title | lowercase | uppercase = ' {{ course.title | lowercase | uppercase }}'

students: '{{ course.students }}'
course.students | number = ' {{ course.students | number }}'

rating: '{{ course.rating }}'
course.rating | number: '2.1-1' = ' {{ course.rating | number: "2.1-1" }}'
course.rating | number: '3.2-2' = ' {{ course.rating | number: "3.2-2" }}'
course.rating | number: '3.1-6' = ' {{ course.rating | number: "3.1-6" }}'

Warning: the currency pipe has been changed in Angular v5. The symbolDisplay option (third parameter) is now a string instead of a boolean. The accepted values are "code", "symbol" or "symbol-narrow".
price: '{{ course.price }}'
course.price | currency = ' {{ course.price | currency }}'
course.price | currency:'AUD':'symbol':'3.2-2' = ' {{ course.price | currency: "AUD":"symbol":"3.2-2" }}'
course.price | currency:'AUD':'symbol-narrow':'3.2-2' = ' {{ course.price | currency: "AUD":"symbol-narrow":"3.2-2" }}'
course.price | currency:'USD':'code':'1.0-0' = ' {{ course.price | currency: "USD":"code":"1.0-0" }}'
course.price | currency:'USD':'symbol-narrow':'1.1-1' = ' {{ course.price | currency: "USD":"symbol-narrow":"1.1-1" }}'

releaseDate: '{{ course.releaseDate }}'
course.releaseDate | date:"shortDate" = ' {{ course.releaseDate | date: "shortDate" }}'
course.releaseDate | date:"short" = ' {{ course.releaseDate | date: "short" }}'
course.releaseDate | date:"medium" = ' {{ course.releaseDate | date: "medium" }}'
course.releaseDate | date:"fullDate" = ' {{ course.releaseDate | date: "fullDate" }}'
